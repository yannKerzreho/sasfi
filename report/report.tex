\section{Setup}

$x_t = \log \RV_t$, z-scored per rolling window $(\mu,\sigma$ frozen until refit$)$.
Direct multi-step: one model per $h \in \{1, 5, 10\}$.
Ridge $\alpha \in \{10^k : k \in [-4,5],\;\Delta k{=}0.25\}$ (37 values), 5-fold CV.

\section{Models}

\subsection{HAR â€” Heterogeneous Autoregressive Model (Corsi 2009)}

\textbf{Features.} Let $\bar{x}^{(w)}_t = \frac{1}{5}\sum_{k=0}^{4}x_{t-k}$
and $\bar{x}^{(m)}_t = \frac{1}{22}\sum_{k=0}^{21}x_{t-k}$.

\begin{equation}
  \hat{x}_{t+h} = \alpha + \beta_d\,x_t + \beta_w\,\bar{x}^{(w)}_t
                         + \beta_m\,\bar{x}^{(m)}_t
\end{equation}

\textbf{HAR-OLS}: $\min_\beta \|X\beta - y\|_2^2$.\quad
\textbf{HAR-Ridge}: column-normalise $[x^{(d)},x^{(w)},x^{(m)}]$; penalise slopes only;
back-transform $w = w_\text{scaled}/\sigma_\text{col}$.

\subsection{NLinear and DLinear (Zeng et al.\ 2023)}

Let $\bm{x}_t = (x_{t-L+1},\ldots,x_t)^\top \in \R^L$ be the lookback window.

\textbf{NLinear.}
\begin{equation}
  \tilde{\bm{x}}_t = \bm{x}_t - x_t, \qquad
  \hat{x}_{t+h} = \tilde{\bm{x}}_t^\top \W + x_t
\end{equation}

\textbf{DLinear.}
\begin{equation}
  \bm{\tau}_t = \mathrm{MA}_k(\bm{x}_t), \quad
  \bm{\varsigma}_t = \bm{x}_t - \bm{\tau}_t, \qquad
  \hat{x}_{t+h} = \begin{bmatrix}\bm{\tau}_t \\ \bm{\varsigma}_t\end{bmatrix}^\top \W
\end{equation}

\subsection{GARCH â€” AR($p$)-GARCH(1,1)}

$x_t = \mu + \sum_{i=1}^{p} \phi_i x_{t-i} + \varepsilon_t$,\quad
$\sigma^2_t = \omega + \alpha_{\scriptscriptstyle G}\varepsilon^2_{t-1} + \beta_{\scriptscriptstyle G}\sigma^2_{t-1}$.
Multi-step: iterate variance recursion analytically. QMLE.

\subsection{SAS â€” State Affine System}

\textbf{Diagonal basis.} $P_0,\ldots,P_p \in \R^n$,\; $Q_0,\ldots,Q_q \in \R^n$, $z\in\mathbb R$.

\vspace{-0.5em}
\begin{align}
  P(z) &= \mathrm{clip}\!\Bigl(\sum_{d=0}^{p} P_d\,z^d,\;
           {-1{+}\epsilon},\;{1{-}\epsilon}\Bigr) \in (-1,1)^n
         \\
  Q(z) &= \sum_{d=0}^{q} Q_d\,z^d \in \R^n \\
  s_t  &= P(z_t) \odot s_{t-1} + Q(z_t)
           \label{eq:sas-rec}
\end{align}

Clipping ensures $|P(z)_i|<1$ (unconditional stability).
Readout: $\hat{x}_{t+h} = s_t^\top \W$, $W_h$ fitted by ridge regression with 5-fold CV.

\textit{Transition polynomial $P$.}  Let $\rho$ be the spectral-norm bound and
$\delta = \max(0.005,\,(1{-}\rho)\cdot 0.3)$ a small headroom margin.
\begin{itemize}\itemsep1pt
  \item $P_0[i] \overset{\text{iid}}{\sim} \mathcal{U}(-\rho,\,\rho)$ â€” base
        eigenvalues spread uniformly to encode timescales from fast-decaying to
        near-unit-root.
  \item $P_k[i] \overset{\text{iid}}{\sim} \mathcal{N}\!\bigl(0,\,\delta/5^k\bigr)$
        for $k \geq 1$ â€” exponentially shrinking input-modulation terms so that
        the clip boundary is rarely reached for $|z| \leq 5$.
\end{itemize}

\textit{Input-drive polynomial $Q$.}  The injection $Q(z_t)$ adds noise with
variance $\propto \sum_k Q_k[i]^2\,\mathbb{E}[z^{2k}]$.  For $z \sim \mathcal{N}(0,1)$,
$\mathbb{E}[z^{2k}] = (2k-1)!! = 1\cdot 3\cdot 5\cdots(2k{-}1)$ (double
factorial).  To keep injection variance roughly uniform across degrees:
\begin{equation}
  Q_k[i] \overset{\text{iid}}{\sim}
  \mathcal{N}\!\!\left(0,\;\frac{1}{\sqrt{n\,(2k-1)!!}}\right)
\end{equation}
so $k{=}0,1$ use scale $1/\!\sqrt{n}$ (unchanged from a linear SAS), while
$k{=}2$ is shrunk by $1/\!\sqrt{3}\approx 0.58$ and $k{=}3$ by
$1/\!\sqrt{15}\approx 0.26$.  This prevents state explosion for large inputs
without hand-tuning per-degree scales.

\textbf{Associative structure.} $\ell_t = (P(z_t), Q(z_t))$. Composition rule:
\begin{equation}
  (A_j,\,b_j) \,\circ\, (A_i,\,b_i) \;=\;
  \bigl(A_j \odot A_i,\;\;A_j \odot b_i + b_j\bigr)
  \label{eq:combine}
\end{equation}
is associative $\Rightarrow$ parallel divide-and-conquer scan.

\begin{algorithm}[H]
\caption{Two-Level Parallel Scan â€” compute $s_1,\ldots,s_T$ from $z_1,\ldots,z_T$}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Transitions $\ell_1,\ldots,\ell_T$;\; chunk size $B$;\; $s_0 = \bm{0}$}
\Output{States $s_1,\ldots,s_T$;\; $s_T$}
$K \leftarrow \lceil T/B \rceil$; pad to $KB$ \\
\textbf{Phase 1} (vmap over $K$ chunks)\;
\quad $\forall\,k:\;(A^\text{intra}_{k,\cdot},\,b^\text{intra}_{k,\cdot})
\leftarrow \textsc{AssocScan}(\circ,\;\ell_{kB+1},\ldots,\ell_{(k+1)B})$ \\
\textbf{Phase 2} (inter-chunk scan over $K$ endpoints)\;
\quad $(A^\text{inter},\,b^\text{inter})
\leftarrow \textsc{AssocScan}(\circ,\;(A^\text{intra}_{1,B},b^\text{intra}_{1,B}),\ldots)$ \\
\textbf{Phase 3} (carry for chunk $k \geq 1$)\;
\quad $c_0 = s_0$;\;
$c_k = A^\text{inter}_{k-1} \odot s_0 + b^\text{inter}_{k-1}$ \\
\textbf{Phase 4} (vmap over $K$ chunks)\;
\quad $\forall\,k,j:\;s_{kB+j} = A^\text{intra}_{k,j} \odot c_k + b^\text{intra}_{k,j}$
\end{algorithm}

Depth $O(\log T)$ vs.\ $O(T)$ sequential.

\subsection{RNN â€” LSTM / GRU / LRU}

Readout: $\hat{x}_{t+h} = \text{Linear}_h(h_T)$, one head per $h$.
Training: Adam, early stopping (patience=20, val=20\% chronological).

\textbf{LRU} (Orvieto et al.\ 2023):
\begin{equation}
  h_t = \sigma(\nu) \odot h_{t-1} + Bz_t, \qquad y_t = Ch_t,
  \qquad \sigma(\nu) \in (0,1)^d
\end{equation}


\section{Evaluation Protocol}

\begin{algorithm}[H]
\caption{Rolling Out-of-Sample Evaluation}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{Series $x_1,\ldots,x_T$; window $W$; refit frequency $R$; horizons $\mathcal{H}$}
\Output{Squared-error sequences $\{e^{(h)}_{t,m}\}$ for each model $m$, horizon $h$}
$({\mu}, {\sigma}) \leftarrow (0, 1)$\;
\For{$t = W$ \KwTo $T - \max(\mathcal{H})$}{
  $\text{train} \leftarrow x_{t-W:t}$\;
  \If{$t \bmod R = 0$}{
    $(\mu,\sigma) \leftarrow \textsc{FitScaler}(\text{train})$\;
    $\tilde{\text{train}} \leftarrow (\text{train} - \mu)/\sigma$\;
    \ForEach{model $m$}{$m.\textsc{Fit}(\tilde{\text{train}},\,\mathcal{H})$}
  }
  $z_t \leftarrow (x_t - \mu)/\sigma$\;
  \ForEach{model $m$, horizon $h \in \mathcal{H}$}{
    $\hat{z}_{t+h} \leftarrow m.\textsc{Predict}(h)$\;
    $e^{(h)}_{t,m} \leftarrow (\hat{z}_{t+h} - z_{t+h})^2$\;
  }
  \ForEach{model $m$}{$m.\textsc{Update}(z_t)$}
}
\end{algorithm}

\textbf{Model Confidence Set (Hansen et al.\ 2011).}
$L \in \R^{T \times M}$, row-normalised by cross-model mean to prevent high-vol dominance.
\begin{enumerate}\itemsep0pt
  \item $d_{t,m} = L_{t,m} - \bar{L}_t$
  \item $T_R = \max_m \bar{d}_m / \hat\sigma_m$ \quad ($\hat\sigma_m$: circular block bootstrap, $B$ reps)
  \item Eliminate worst model if $p < \alpha$; repeat until $H_0$ not rejected.
  \item Surviving set = MCS$_\alpha$.
\end{enumerate}

\textbf{MCS Frequency.} Run MCS per symbol per horizon; report $c_m^{(h)}/N$.

\subsection{SAS as Data-Generating Process â€” Specification Experiment}

\textbf{DGP construction.}
For each real-data symbol, we fit SAS\textsubscript{q2} on the \emph{full}
real series (all $T$ observations) to obtain initialised reservoir matrices
$(P,Q)$, readout $W_1$, terminal state $s_T$, and residual noise scale
$\hat\sigma = \mathrm{std}(z_{t+1} - s_t^\top W_1)_{t=\tau}^{T-1}$.
We then generate $N_\text{gen}{=}4{,}000$ synthetic steps in closed-loop
autoregressive form:

\begin{algorithm}[H]
\caption{SAS-DGP autoregressive generation}
\SetKwInOut{Input}{Input}
\Input{$(P,Q,W_1,s_T,\hat\sigma)$;\; $\nu$ (Student-$t$ df)}
$s \leftarrow s_T$\;
\For{$t = 1$ \KwTo $N_\text{gen}$}{
  $\varepsilon \sim t(\nu)$\;
  $z_t \leftarrow s^\top W_1 + \hat\sigma\,\varepsilon$\;
  $s   \leftarrow P(z_t)\odot s + Q(z_t)$\;
}
\end{algorithm}

No output clipping is applied: \texttt{DiagonalPoly} always clips $P(z)$ to
$(-0.9999,\,0.9999)$ element-wise, making the state recurrence unconditionally
contractive for any $z$.
This is essential for oracle correctness â€” clipping would introduce a Jensen-gap
between the stored conditional mean $s_t^\top W_1$ and the true
$\mathbb{E}[\mathrm{clip}(s_t^\top W_1 + \hat\sigma\varepsilon)]$,
allowing adaptive models to beat the oracle.
The raw (un-normalised) synthetic series is used for OOS evaluation
($W{=}2000$, $R{=}20$, $h{=}1$), preserving the same $z$-space as the DGP.

Eight models are evaluated: the \textbf{Noiseless oracle} (DGP's fixed $W_1$,
exact conditional mean $s_t^\top W_1$), \texttt{SAS\_same\_fix\-ed$\lambda$}
(same reservoir as DGP, $\lambda$ fixed to DGP value), \texttt{SAS\_same\_cv$\lambda$}
(same reservoir, rolling CV for $\lambda$), \texttt{SAS\_diff} (different seed,
same $q{=}2$), \texttt{SAS\_q1} ($q{=}1$ misspecification), and HAR, DLinear, GARCH.
Each symbol contributes three independent runs (\texttt{DGP\_SEEDS}$=[42,137,256]$).

\textbf{Findings.}
\begin{itemize}
  \item \emph{Oracle floor confirmed.}  Noiseless achieves the lowest mean MSE
        ($0.5084$), consistent with the theoretical floor $\hat\sigma^2\cdot 5/3$.
        All other models sit strictly above it.
  \item \emph{Fixed $\lambda$ near-optimal.}  \texttt{SAS\_same\_fixed$\lambda$}
        ($0.5127$) beats \texttt{SAS\_same\_cv$\lambda$} ($0.5129$): the DGP's
        $\lambda$ transfers well to the synthetic series (median oracle $\lambda$
        is only $+0.25$ log$_{10}$ away from DGP $\lambda$), so CV adds noise
        without improving the mean.
  \item \emph{Reservoir seed matters little.}  \texttt{SAS\_diff} ($0.5173$)
        is only $0.005$ above \texttt{SAS\_same\_fixed$\lambda$}: any $q{=}2$
        reservoir of $n{=}50$ units recovers nearly the same MSE floor.
  \item \emph{MCS bias--variance tradeoff.}  Noiseless has the best mean MSE
        but only 54\% MCS survival, while adaptive SAS models reach 86--88\%.
        The oracle's pure $t(5)$ residuals carry kurtosis~6: squared losses are
        very heavy-tailed, making per-symbol win margins statistically fragile.
        Adaptive models reduce effective residual kurtosis by partially fitting
        the signal, yielding more stable pairwise loss differences in the MCS test.
\end{itemize}
